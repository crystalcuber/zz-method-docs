import { useRouter } from "next/router";
import Script from "next/script";
import "./styles.css";
import { Analytics } from "@vercel/analytics/react";
import { useEffect } from "react";

export default function App({ Component, pageProps }) {
  // GoatCounter analytics integration, taken from https://github.com/arp242/goatcounter/issues/482#issuecomment-1016033292
  const router = useRouter();
  useEffect(
    function sendGoatCounterEventsOnRoute() {
      const handleRouteChange = (path) => {
        window?.goatcounter?.count?.({
          path,
        });
      };
      router.events.on("routeChangeComplete", handleRouteChange);
      return () => {
        router.events.off("routeChangeComplete", handleRouteChange);
      };
    },
    [router.events]
  );
  // TODO: remove Vercel Analytics, plan ran out
  return (
    <>
      <Component {...pageProps} />
      <Analytics />
      <Script
        async
        data-goatcounter="https://err0rcuber.goatcounter.com/count"
        src="//gc.zgo.at/count.js"
        strategy="afterInteractive"
      />
    </>
  );
}
